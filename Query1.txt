Select 

IFA01.CONS_INTAKE_FORM.CONS_INTAKE_FORM_ID,
IFA01.CONS_INTAKE_FORM.PROJ_NM,
IFA01.CONS_INTAKE_FORM.BL_ID,
IFA01.CONS_INTAKE_FORM.CRTE_USR_ID,
IFA01.CONS_INTAKE_FORM.CRTE_TSTP,
IFA01.CONS_INTAKE_FORM.REQ_RLSE_DT,

IFA01.STAT.STAT_DESC,

IFA01.INTAKE_IMPACTED_TOOL.INTAKE_IMPACTED_TOOL_TYP_CD,

IFA01.PROJ_TYP.PROJ_TYP_DESC

FROM IFA01.CONS_INTAKE_FORM

INNER JOIN IFA01.CONS_INTAKE_FORM_STAT ON IFA01.CONS_INTAKE_FORM_STAT.CONS_INTAKE_FORM_ID = IFA01.CONS_INTAKE_FORM.CONS_INTAKE_FORM_ID

INNER JOIN IFA01.STAT ON IFA01.STAT.STAT_CD = IFA01.CONS_INTAKE_FORM_STAT.INTAKE_HLTH_STAT_CD;

INNER JOIN IFA01.INTAKE_IMPACTED_TOOL ON IFA01.INTAKE_IMPACTED_TOOL.CONS_INTAKE_FORM_ID = IFA01.CONS_INTAKE_FORM.CONS_INTAKE_FORM_ID

INNER JOIN IFA01.PROJ_TYP ON IFA01.PROJ_TYP.PROJ_TYP_CD = IFA01.CONS_INTAKE_FORM.PROJ_TYP_CD

WHERE

IFA01.CONS_INTAKE_FORM.CRTE.USR_ID=logged in user OR IFA01.CONS_INTAKE_FORM.CONS_INTAKE_FORM_CNTC.CNTC_NM = logged in user
