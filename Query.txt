Select a.CONS_INTAKE_FORM_ID,
a.PROJ_NM,
a.BL_ID,
a.CRTE_USR_ID,
a.CRTE_TSTP,
a.REQ_RLSE_DT,
a.PROJ_TYP_CD,

b.INTAKE_HLTH_STAT_CD,

c.STAT_DESC,

d.INTAKE_IMPACTED_TOOL_TYP_CD,

e.PROJ_TYP_DESC

FROM IFA01.CONS_INTAKE_FORM a,
IFA01.CONS_INTAKE_FORM_STAT b,
IFA01.STAT c,
IFA01.INTAKE_IMPACTED_TOOL d,
IFA01.PROJ_TYP e

WHERE

a.CRTE.USR_ID=logged in user OR aCONS_INTAKE_FORM_CNTC.CNTC_NM = logged in user
AND
a.CONS_INTAKE_FORM_ID = b.CONS_INTAKE_FORM_ID
AND
b.INTAKE_HLTH_STAT_CD = c.STAT_CD
AND
a.CONS_INTAKE_FORM_ID = d.CONS_INTAKE_FORM_ID
AND
a.PROJ_TYP_CD = e.PROJ_TYP_CD
